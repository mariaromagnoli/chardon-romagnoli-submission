\begin{frame}{Results - \textit{specification 1}}
\alert{\textbf{Spec. 1: DDD}} \\
\vspace{0.2cm}
\textcolor{violet}{\textbf{Variation:}} comparing wage changes for different types of workers.  Across fap*region that became listed after April 2021 vs. those that did not.

\begin{align*}
\ln(w_{ijrt}) &= \alpha + \beta_1 \text{EEA}_{i} + \beta_2 \text{NonEEA}_{i} + \beta_3 \text{post}_{t} + \mathbf{X}_{ijrt}'\boldsymbol{\gamma} \\ &+ \beta_4 (\text{French}_{i} \times \text{treatment}_{jr} \times \text{post}_{t}) + \beta_5 (\text{EEA}_{i} \times \text{treatment}_{jr} \times \text{post}_{t}) \\ & + \beta_6 (\text{NonEEA}_{i} \times \text{treatment}_{jr} \times \text{post}_{t}) + \delta_{jr} + \delta_{t} + \varepsilon_{ijrt}
\end{align*}
\vspace{-0.2cm}
\begin{itemize}
    \item \alert{$\beta_1, \beta_2$}: baseline wage gap, unconditional on listing.
    \item \alert{$\beta_3$}: common wage trend after April 2021.
    \item \alert{$\beta_4$}: wage change for French workers in listed after Apr2021.
    \item \alert{$\beta_5$}: differential wage change for EEA workers in listed after Apr2021 (\textit{should be 0}).
    \item \alert{$\beta_6$}: differential wage change for non-EEA workers in listed cells after Apr2021 (\textit{should be negative}).
\end{itemize}
\end{frame}

\section*{Appendix}

\begin{frame}{Appendix - \textit{data cleaning procedure}}
\label{datacleaning}

\textbf{\alert{Ficher Historique}} $\rightarrow$ \textit{socio-demographic data}.
\begin{itemize}
	\item Filter: relevant nationalities, relevant period, relevant variables.
	\item Create: eea/non-eea and sample dummies.
	\item Harmonize: if socio-demographic change over time, take latest value.
\end{itemize}
\vspace{0.4cm}
\textbf{\alert{Mouvements de Main d'Œuvre}} $\rightarrow$ \textit{wage data}.
\begin{itemize}
	\item Filter: relevant people, contracts after 2019, longer than 30 days.
	\item Create: period dummies, extract the region of the contract from department codes.
	\item Harmonize: wages. 
\end{itemize}
\vspace{0.2cm}
\textbf{\alert{Shortage lists}} $\rightarrow$ \textit{treatment data}.
\begin{itemize}
	\item Scrape + clean shortage list.
	\item Map FAP codes (in the policy) to PCS codes (in MMO).
\end{itemize}
